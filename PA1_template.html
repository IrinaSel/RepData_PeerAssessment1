dataset$day <- weekdays(dataset$date)

    if (dataset$day[i] %in% c("Saturday","Sunday")) {             
        dataset$day[i]<-"weekend"                                 
    }
    else{
        dataset$day[i]<-"weekday"                                 
    }
}
